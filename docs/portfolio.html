Portfolio

returns_monthly <- c("URTH", "EEM", "^NDX") %>%
    tq_get(get  = "stock.prices",
           from = "2015-01-01",
           to   = today) %>%
    group_by(symbol) %>%
    tq_transmute(select     = adjusted, 
                 mutate_fun = periodReturn, 
                 period     = "monthly", 
                 col_rename = "Ra")
returns_monthly
weights <- c(
    0.75, 0.25, 0.0,
    1.00, 0.00, 0.00,
    0.00, 0.00, 1.00
)
stocks <- c("URTH", "EEM", "^NDX")
weights_table <-  tibble(stocks) %>%
    tq_repeat_df(n = 3) %>%
    bind_cols(tibble(weights)) %>%
    group_by(portfolio)
weights_table
returns_monthly_multi <- returns_monthly %>%
    tq_repeat_df(n = 3)
returns_monthly_multi
portfolio_returns_monthly_multi <- returns_monthly_multi %>%
    tq_portfolio(assets_col  = symbol, 
                 returns_col = Ra, 
                 weights     = weights_table, 
                 col_rename  = "Ra")
portfolio_returns_monthly_multi
portfolio_growth_monthly_multi <- returns_monthly_multi %>%
    tq_portfolio(assets_col   = symbol, 
                 returns_col  = Ra, 
                 weights      = weights_table, 
                 col_rename   = "investment.growth",
                 wealth.index = TRUE) %>%
    mutate(investment.growth = investment.growth * 10000)
